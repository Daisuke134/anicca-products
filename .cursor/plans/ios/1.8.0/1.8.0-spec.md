**Trial period テキスト生成（動的）:**

```swift
private var trialPeriodText: String? {
    guard let discount = selectedPackage?.storeProduct.introductoryDiscount else { return nil }
    let period = discount.subscriptionPeriod
    let isJA = Locale.current.languageCode == "ja"
    switch period.unit {
    case .day:   return isJA ? "\(period.value)日間" : period.value == 1 ? "1 Day" : "\(period.value) Days"
    case .week:  return isJA ? "\(period.value)週間" : period.value == 1 ? "Week" : "\(period.value) Weeks"
    case .month: return isJA ? "\(period.value)ヶ月" : period.value == 1 ? "1 Month" : "\(period.value) Months"
    case .year:  return isJA ? "\(period.value)年" : period.value == 1 ? "1 Year" : "\(period.value) Years"
    @unknown default: return nil
    }
}
```
