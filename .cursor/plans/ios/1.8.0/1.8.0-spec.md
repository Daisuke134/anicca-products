**Trial period テキスト生成（動的）:**

```swift
private var trialPeriodText: String? {
    guard let discount = selectedPackage?.storeProduct.introductoryDiscount else { return nil }
    let period = discount.subscriptionPeriod
    let isJA = Locale.current.languageCode == "ja"
    switch period.unit {
    case .day:   return isJA ? "\(period.value)日間" : period.value == 1 ? "1 Day" : "\(period.value) Days"
    case .week:  return isJA ? "\(period.value)週間" : period.value == 1 ? "Week" : "\(period.value) Weeks"
    case .month: return isJA ? "\(period.value)ヶ月" : period.value == 1 ? "1 Month" : "\(period.value) Months"
    case .year:  return isJA ? "\(period.value)年" : period.value == 1 ? "1 Year" : "\(period.value) Years"
    @unknown default: return nil
    }
}
```

#### 8.7.3 ペイウォール A/B 週次レポート Cron

**スキル名:** `posthog-paywall-report`
**Cron:** 毎週月曜 10:00 JST（`0 10 * * 1` tz=Asia/Tokyo）
**実行環境:** Mac Mini（OpenClaw Gateway）

**処理フロー:**

```
Step 1: PostHog API からデータ取得
  curl -H "Authorization: Bearer $POSTHOG_PERSONAL_API_KEY" \
    "https://us.posthog.com/api/projects/327882/experiments/364239/"
  
  取得する値:
  ├── results.control.count        （Variant A の paywall_viewed 数）
  ├── results.control.conversion   （Variant A の paywall_purchased 数）
  ├── results.test.count           （Variant B の paywall_viewed 数）
  ├── results.test.conversion      （Variant B の paywall_purchased 数）
  └── results.significance         （統計的有意性 0-100%）

Step 2: CVR 計算
  control_cvr = control.conversion / control.count * 100
  test_cvr    = test.conversion / test.count * 100
  lift        = (test_cvr - control_cvr) / control_cvr * 100

Step 3: 判定ロジック
  ├── significance >= 95% AND サンプル各100+  → 「✅ 勝者確定」
  ├── significance 80-95%                      → 「⏳ 継続。サンプル不足」
  ├── significance < 80% AND 4週間経過          → 「⚠️ 差なし。デザイン変更推奨」
  └── significance < 80% AND 4週間未満          → 「⏳ 継続」

Step 4: Slack #metrics に投稿
  📊 Paywall A/B Report — Week {N}
  ━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
  Variant A (control): {control_cvr}% CVR ({conversion}/{viewed})
  Variant B (test):    {test_cvr}% CVR ({conversion}/{viewed})
  Lift: {lift}%
  Significance: {significance}%
  判定: {判定結果}
  ━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
  Dashboard: https://us.posthog.com/project/327882/experiments/364239
```
